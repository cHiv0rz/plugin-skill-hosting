   https://{{ $host.host }}/
{{- end }}
{{- else }}
   kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "psh.frontend.fullname" . }} 8080:{{ .Values.frontend.service.port }}
{{- end }}

2. Marketplace URL to give to Claude Code users:
   /plugin marketplace add {{ .Values.publicBaseURL }}/marketplace.json

   Note: Claude Code rejects plain http:// plugin sources. publicBaseURL MUST
   be https://; the ingress TLS secret ({{ (index .Values.ingress.tls 0).secretName }})
   covers that.

3. Application secret. The chart does NOT create one — apply your own
   Secret (or SealedSecret) named "{{ include "psh.secretName" . }}" in
   namespace "{{ .Release.Namespace }}" before/while installing.
   Override the name with --set existingSecret=<your-secret>.

   kubectl create secret generic {{ include "psh.secretName" . }} \
     --namespace {{ .Release.Namespace }} \
     --dry-run=client -o yaml \
{{- if .Values.postgres.enabled }}
     --from-literal=POSTGRES_PASSWORD=<db-password> \
     --from-literal=JWT_SECRET=<32+chars> \
{{- else }}
     --from-literal=DATABASE_URL='postgres://user:pass@host:5432/db?sslmode=require' \
     --from-literal=JWT_SECRET=<32+chars> \
{{- end }}
     --from-literal=API_TOKEN_ENC_KEY=<optional, generate once: openssl rand -hex 32> \
     | kubeseal --format yaml | kubectl apply -f -

   API_TOKEN_ENC_KEY is optional (derived from JWT_SECRET if omitted). If you set
   it, generate it ONCE and keep it stable — re-running this with a fresh value
   re-keys token display (auth is unaffected; tokens just need regenerating).
